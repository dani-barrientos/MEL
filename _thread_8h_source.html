221 template<class T> ::mel::core::WaitResult<T> waitForFutureThread( const mel::core::Future<T>& f,unsigned int msecs = ::mel::core::Event::EVENT_WAIT_INFINITE)
222 {
223 using ::mel::core::Event;
224 struct _Receiver
225 {
226 _Receiver():mEvent(false,false){}
227 using futT = mel::core::Future<T>;
228 mel::core::EWaitError wait( const futT& f,unsigned int msecs)
229 {
230 Event::EWaitCode eventresult;
231 int evId = f.subscribeCallback(
232 std::function<::mel::core::ECallbackResult( typename futT::ValueType&)>([this](typename futT::ValueType& )
233 {
234 mEvent.set();
235 return ::mel::core::ECallbackResult::UNSUBSCRIBE;
236 }));
237 eventresult = mEvent.wait(msecs);
238 f.unsubscribeCallback(evId);
239 switch( eventresult )
240 {
241 case ::mel::core::Event::EVENT_WAIT_OK:
242 return ::mel::core::EWaitError::FUTURE_WAIT_OK;
243 break;
244 case ::mel::core::Event::EVENT_WAIT_TIMEOUT:
245 return ::mel::core::EWaitError::FUTURE_WAIT_TIMEOUT;
246 break;
247 case ::mel::core::Event::EVENT_WAIT_ERROR:
248 return ::mel::core::EWaitError::FUTURE_UNKNOWN_ERROR;
249 break;
250 default:
251 return ::mel::core::EWaitError::FUTURE_WAIT_OK; //silent warnin
252 }
253 }
254 private:
255 mel::core::Event mEvent;
256 };
257 auto receiver = std::make_unique<_Receiver>();
258 mel::core::EWaitError waitRes = receiver->wait(f,msecs);
259 switch (waitRes)
260 {
261 case ::mel::core::EWaitError::FUTURE_WAIT_OK:
262 if ( f.getValue().isValid())
263 return ::mel::core::WaitResult(f);
264 else
265 std::rethrow_exception(f.getValue().error());
266 break;
267 case ::mel::core::EWaitError::FUTURE_RECEIVED_KILL_SIGNAL:
268 throw mel::core::WaitException(mel::core::EWaitError::FUTURE_RECEIVED_KILL_SIGNAL,"Kill signal received");
269 break;
270 case ::mel::core::EWaitError::FUTURE_WAIT_TIMEOUT:
271 throw mel::core::WaitException(mel::core::EWaitError::FUTURE_RECEIVED_KILL_SIGNAL,"Time out exceeded");
272 break;
273 case ::mel::core::EWaitError::FUTURE_UNKNOWN_ERROR:
274 throw mel::core::WaitException(mel::core::EWaitError::FUTURE_UNKNOWN_ERROR,"Unknown error");
275 break;
276 default:
277 throw mel::core::WaitException(mel::core::EWaitError::FUTURE_UNKNOWN_ERROR,"Unknown error");
278 break;
279 }
