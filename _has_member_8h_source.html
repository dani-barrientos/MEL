6 /*
7 * macros and template to know if an object has a specific member function
8 * usage. Example: suppose you want to know if some Class has a member function int pepito( float,char) const
9 * Then you must use macro DECLARE_HASMEMBER as follows:
10 * DECLARE_HASMEMBER( pepito, int,(float,char),const );
11 * First parameter is name function, second is return type, third are functions parameters and fourth is any attribute (const, throw,...)
12 * This macro create a new template struct called Haspepito
13 *
14 * Once declared this new struct, you can use it. If you have a class called MiClase, you will use it as follows:
15 * Haspepito<MiClase>::Result
16 * This is a compiler-time bool constant that you can use to decide to, for example, to use or no to use that function, or execute
17 * a different program path, etc
18 *
19 * @remarks THIS IS NOT ONLY VALID( yet ) FOR MEMBERS BELONGING TO PARENT CLASSES. So if,in the previous example, MiClase inherits from
20 * MiClase_Base and this class has pepito method then, Haspepito<MiClase>::Result is FALSE (this will be addressed in a future)
21 */
22 #define DECLARE_HASMEMBER(func_name,TRet,args, attributes ) \
23 template <class T> struct Has##func_name \
24 { \
25 template <class U,TRet (U::*) args attributes > struct AUXSTRUCT{};\
26 class Big{ char dummy[2];}; \
27 typedef char Small; \
28 template <class U> static Small check( AUXSTRUCT< U,&U::func_name >* ); \
29 template <class U> static Big check( ... ); \
30 enum{ Result = sizeof( check<T>(0)) == sizeof(Small) }; \
31 };
32 }
