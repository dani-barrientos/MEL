218  template <class I, class F> ::mel::parallelism::Barrier Executor<Runnable>::loop(I&& begin, I&& end, F&& functor, int increment)
219  {
220  typedef typename std::decay<I>::type DecayedIt;
221  constexpr bool isArithIterator = mel::mpl::TypeTraits<DecayedIt>::isArith;
222  if ( getRunnable().expired())
223  {
224  throw std::runtime_error("Runnable has been destroyed");
225  }
226  bool autoKill = getOpts().autoKill;
227  bool independentTasks = getOpts().independentTasks;
228  int length;
229  if constexpr (isArithIterator)
230  length = (end-begin);
231  else
232  length = std::distance(begin, end);
233  int nElements = independentTasks?(length+increment-1)/increment:1; //round-up
234  auto ptr = getRunnable().lock();
235 
236  ::mel::parallelism::Barrier barrier(nElements);
237  if ( independentTasks)
238  {
239  for(auto i = begin; i < end;i+=increment)
240  {
241  ptr->fireAndForget(
242  [functor,barrier,i]() mutable noexcept((std::is_nothrow_invocable<F,I>::value))
243  {
244  functor(i);
245  barrier.set();
246  },0,autoKill?Runnable::killTrue:Runnable::killFalse
247  );
248  }
249  }else
250  {
251  ptr->fireAndForget(
252  [functor,barrier,begin,end,increment]() mutable noexcept(std::is_nothrow_invocable<F,I>::value)
253  {
254  for(auto i = begin; i < end;i+=increment)
255  {
256  functor(i);
257  }
258  barrier.set();
259  },0,autoKill?Runnable::killTrue:Runnable::killFalse
260  );
261  }
262  return barrier;
263  }
