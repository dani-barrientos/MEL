88  template <class ExecutorAgent,class TArg,class TRet>
89  ExFuture<ExecutorAgent,typename std::remove_cv<typename std::remove_reference<TRet>::type>::type> inmediate( ExFuture<ExecutorAgent,TArg> fut,TRet&& arg)
90  {
91  static_assert( !std::is_lvalue_reference<TRet>::value ||
92  std::is_const< typename std::remove_reference<TRet>::type>::value,"execution::inmediate. Use std::ref() to pass argument as reference");
93  using NewType = typename std::remove_cv<typename std::remove_reference<TRet>::type>::type;
94  typedef typename ExFuture<ExecutorAgent,TArg>::ValueType ValueType;
95  ExFuture<ExecutorAgent,NewType> result(fut.agent);
96  fut.subscribeCallback(
97  [fut,result,arg = std::forward<TRet>(arg)]( ValueType& input) mutable
98  {
99  //launch tasks as response for callback for two reasons: manage the case when Future is already available when checked, so callback is trigered
100  //on calling thread and to decouple tasks and no saturate execution resource and independence tasks
101  if ( input.isValid() )
102  {
103  launch(fut.agent,[result,arg = std::forward<TRet>(arg)]() mutable noexcept
104  {
105  result.setValue(std::forward<TRet>(arg));
106  });
107  }
108  else
109  {
110  //set error as task in executor
111  launch(fut.agent,[result,err = std::move(input.error())]( ) mutable noexcept
112  {
113  result.setError(std::move(err));
114  });
115  }
116  }
117  );
118  return result;
119  }

221  template <class ExecutorAgent,class TArg, class I, class F> ExFuture<ExecutorAgent,TArg> loop(ExFuture<ExecutorAgent,TArg> source,I&& begin, I&& end, F&& functor, int increment = 1)
222  {
223  static_assert( std::is_invocable<F,int,TArg>::value, "execution::loop bad functor signature");
224  ExFuture<ExecutorAgent,TArg> result(source.agent);
225  typedef typename ExFuture<ExecutorAgent,TArg>::ValueType ValueType;
226  source.subscribeCallback(
227  [source,functor = std::forward<F>(functor),result,begin = std::forward<I>(begin),end = std::forward<I>(end),increment](ValueType& input) mutable
228  {
229  try
230  {
231  if ( input.isValid() )
232  {
233  std::exception_ptr* except = new std::exception_ptr(nullptr);
234  ::mel::parallelism::Barrier barrier;
235  if constexpr (std::is_nothrow_invocable<F,I,TArg>::value)
236  {
237  barrier = source.agent.loop(std::forward<I>(begin), std::forward<I>(end),
238  [f = std::forward<F>(functor),source](I idx) mutable noexcept
239  {
240  //@todo arreglar el loop para que reciba I&&
241  //f(std::forward<I>(idx),fut.getValue());
242 
243  f(idx,source.getValue().value());
244  }
245  , increment);
246  }else
247  {
248  barrier = source.agent.loop(std::forward<I>(begin), std::forward<I>(end),
249  [f = std::forward<F>(functor),source,except](I idx) mutable
250  {
251  try
252  {
253  f(idx,source.getValue().value());
254  }catch(...)
255  {
256  if (!*except)
257  *except = std::current_exception();
258  }
259  }
260  , increment);
261  }
262  barrier.subscribeCallback(
263  std::function<::mel::core::ECallbackResult( const ::mel::parallelism::BarrierData&)>([result,source,except](const ::mel::parallelism::BarrierData& ) mutable
264  {
265  if ( *except ) //any exception?
266  result.setError(*except);
267  else
268  //result.assign(std::move(source.getValue())); //@todo it's not correct, but necesary to avoid a lot of copies. I left this way until solved in the root. Really is not very worrying
269  result.assign(source);
270  delete except;
271  return ::mel::core::ECallbackResult::UNSUBSCRIBE;
272  }));
273  }else
274  {
275  //set error as task in executor
276  launch(source.agent,[result,err = std::move(input.error())]( ) mutable noexcept
277  {
278  result.setError(std::move(err));
279  });
280  }
281  }
282  catch(...)
283  {
284  result.setError( std::current_exception() );
285  }
286  });
287  return result;
288  }
