87  template <class ExecutorAgent,class TArg,class TRet>
88  ExFuture<ExecutorAgent,typename std::remove_cv<typename std::remove_reference<TRet>::type>::type> inmediate( ExFuture<ExecutorAgent,TArg> fut,TRet&& arg)
89  {
90  static_assert( !std::is_lvalue_reference<TRet>::value ||
91  std::is_const< typename std::remove_reference<TRet>::type>::value,"execution::inmediate. Use std::ref() to pass argument as reference");
92  using NewType = typename std::remove_cv<typename std::remove_reference<TRet>::type>::type;
93  typedef typename ExFuture<ExecutorAgent,TArg>::ValueType ValueType;
94  ExFuture<ExecutorAgent,NewType> result(fut.agent);
95  fut.subscribeCallback(
96  std::function<void( ValueType&)>([fut,result,arg = std::forward<TRet>(arg)]( ValueType& input) mutable
97  {
98  //launch tasks as response for callback for two reasons: manage the case when Future is already available when checked, so callback is trigered
99  //on calling thread and to decouple tasks and no staturate execution resoruce and independence tasks
100  //hasta aqu√≠ las copias tienen sentido
101  if ( input.isValid() )
102  {
103  launch(fut.agent,[result,arg = std::forward<TRet>(arg)]() mutable noexcept
104  {
105  result.setValue(std::forward<TRet>(arg));
106  });
107  }
108  else
109  {
110  //set error as task in executor
111  launch(fut.agent,[result,err = std::move(input.error())]( ) mutable noexcept
112  {
113  result.setError(std::move(err));
114  });
115  }
116  })
117  );
118  return result;
119  }

221  template <class ExecutorAgent,class TArg, class I, class F> ExFuture<ExecutorAgent,TArg> loop(ExFuture<ExecutorAgent,TArg> source,I&& begin, I&& end, F&& functor, int increment = 1)
222  {
223  ExFuture<ExecutorAgent,TArg> result(source.agent);
224  typedef typename ExFuture<ExecutorAgent,TArg>::ValueType ValueType;
225  source.subscribeCallback(
226  std::function<void( ValueType&)>([source,functor = std::forward<F>(functor),result,begin = std::forward<I>(begin),end = std::forward<I>(end),increment](ValueType& input) mutable
227  {
228  try
229  {
230  if ( input.isValid() )
231  {
232  std::exception_ptr* except = new std::exception_ptr(nullptr);
233  ::mel::parallelism::Barrier barrier;
234  if constexpr (std::is_nothrow_invocable<F,I,TArg&>::value)
235  {
236  barrier = source.agent.loop(std::forward<I>(begin), std::forward<I>(end),
237  [f = std::forward<F>(functor),source](I idx) mutable noexcept
238  {
239  //@todo arreglar el loop para que reciba I&&
240  //f(std::forward<I>(idx),fut.getValue());
241 
242  f(idx,source.getValue().value());
243  }
244  , increment);
245  }else
246  {
247  barrier = source.agent.loop(std::forward<I>(begin), std::forward<I>(end),
248  [f = std::forward<F>(functor),source,except](I idx) mutable
249  {
250  try
251  {
252  f(idx,source.getValue().value());
253  }catch(...)
254  {
255  if (!*except)
256  *except = std::current_exception();
257  }
258  }
259  , increment);
260  }
261  barrier.subscribeCallback(
262  std::function<::mel::core::ECallbackResult( const ::mel::parallelism::BarrierData&)>([result,source,except](const ::mel::parallelism::BarrierData& ) mutable
263  {
264  if ( *except ) //any exception?
265  result.setError(*except);
266  else
267  //result.assign(std::move(source.getValue())); //@todo it's not correct, but necesary to avoid a lot of copies. I left this way until solved in the root. Really is not very worrying
268  result.assign(source);
269  delete except;
270  return ::mel::core::ECallbackResult::UNSUBSCRIBE;
271  }));
272  }else
273  {
274  //set error as task in executor
275  launch(source.agent,[result,err = std::move(input.error())]( ) mutable noexcept
276  {
277  result.setError(std::move(err));
278  });
279  }
280  }
281  catch(...)
282  {
283  result.setError( std::current_exception() );
284  }
285  }));
286  return result;
287  }
