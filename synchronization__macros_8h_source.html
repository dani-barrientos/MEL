71 Future<TRet> result = runnable->execute<TRet>( CREATE_LINKER( TRet,function_name##_sync,args ) ); \
72 result.wait();\
73 if ( !result.getValid() ){\
74 if ( result.getError()->error == Runnable::ERRORCODE_EXCEPTION ){\
75 Runnable::ExecuteErrorInfo* ei = (Runnable::ExecuteErrorInfo*)result.getError();\

79 }else \

81 }else return result.getValue();}
